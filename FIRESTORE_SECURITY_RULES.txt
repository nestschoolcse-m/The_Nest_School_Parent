/**
 * FIRESTORE SECURITY RULES
 * 
 * Apply these rules to your Firestore database via Firebase Console
 * 
 * Path: Firestore > Rules
 * 
 * These rules:
 * 1. Prevent direct client-side writes to attendance_logs
 * 2. Only allow server-side API writes (via service account)
 * 3. Allow authenticated users to read their own attendance
 * 4. Prevent unauthorized access
 */

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Default: Deny all access
    match /{document=**} {
      allow read, write: if false;
    }

    // Attendance logs collection - Server-side writes only
    match /attendance_logs/{docId} {
      // Deny direct client writes
      // Only server (API route) can write via service account
      allow write: if request.auth == null || false;
      
      // Allow authenticated users to read attendance records
      // In production, filter by user's ward USN
      allow read: if request.auth != null;
      
      // Validate attendance_logs document structure on server
      allow write: if request.resource.data.keys().hasAll([
        'usn',
        'wardName',
        'type',
        'timestamp',
      ]) && 
      request.resource.data.type in ['ENTRY', 'EXIT'] &&
      request.resource.data.usn is string &&
      request.resource.data.wardName is string;
    }

    // Parent credentials collection - For auth use only
    match /parentCredentials/{usn} {
      // Only allow service account (admin) writes
      allow write: if false;
      allow read: if false;
    }

    // Students collection (from dataDb) - Read-only
    match /students/{usn} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // Additional rules for future features
    match /notifications_queue/{docId} {
      allow read, write: if false; // Reserved for backend
    }

    match /user_devices/{userId}/{docId} {
      allow read: if request.auth.uid == userId;
      allow write: if false; // Backend only
    }
  }
}
